<b>This map shows data of indicator 15.4.1, the Average proportion of Mountain Key Biodiversity Areas (KBAs) covered by protected areas (%). This is a part of SDGs goal 15: Life on Land. One of the targets is target 15.4, ENSURE CONSERVATION OF MOUNTAIN ECOSYSTEMS. The UN website explains, “By 2030, ensure the conservation of mountain ecosystems, including their biodiversity, in order to enhance their capacity to provide benefits that are essential for sustainable development.” Here is the link to the SDG indicator metadata: (https://unstats.un.org/sdgs/metadata/files/Metadata-15-05-01.pdf ). <br>
A choropleth map shows the distribution of data representing geographic space in colors or patterns. It helps readers to analyze the geographic distribution of data. However, the data representing the area can change depending on the grouping of areas on the map. Therefore, this method is useful to see broad distribution but is weak at spotting specific points in detail. <br>
While making this map, I learned how to make a Choropleth Map in QGIS. It started with importing data from websites and cleaning the data. To make a Choropleth map, I need to save the data in different layers. Learning the skills to design and modify the map was easy when I knew what kind of data and folders I was handling and what kind of commands I was giving the computer. <br>

<script type="text/javascript">
  // Define the geographic bounding box for your project area.
  // Replace these coordinates with the bounding box of your study area.
  // Format for Leaflet LatLng: [lat, lng]
  const southWest = L.latLng(51.0, 3.5);   // SW corner (lat, lng) — adjust
  const northEast = L.latLng(53.0, 6.5);   // NE corner (lat, lng) — adjust
  const bounds = L.latLngBounds(southWest, northEast);

  // Create the map with zoom and bounds limits
  const map = L.map('map', {
    center: [52.1, 4.9],    // initial center [lat, lng] — keep your preferred center
    zoom: 2,               // initial zoom
    minZoom: 1,             // prevents zooming out too far (increase to restrict more)
    maxZoom: 18,            // prevents zooming in too far
    maxBounds: bounds,      // lock panning to the bounding box
    maxBoundsViscosity: 0.9 // 0 = free, 1 = fully constrained; 0.9 is strongly constrained
  });

//UPDATE above - notice the SDG15v3.js file - that is my geometry file - replace that file with your own!
//call the basemap you want to use
	const tiles = L.tileLayer('https://tile.openstreetmap.org/{z}/{x}/{y}.png', {
		maxZoom: 19,
		attribution: '&copy; <a href="http://www.openstreetmap.org/copyright">OpenStreetMap</a>'
	}).addTo(map);

	// control that shows attribute info on hover
	const info = L.control();

	info.onAdd = function (map) {
		this._div = L.DomUtil.create('div', 'info');
		this.update();
		return this._div;
	};

		//UPDATE to call your own data for the interactivity- the name after "props" is the attribute name - look in your js file to find the attribute to call
		// change class breaks and color depending on your values //UPDATE for our own legend - use colorbrewer.org for ideas for colors
		//UPDATE legend to match the classes you coded - and the source of your data needs to reference the correct link
	info.update = function (props) {
		const contents = props ? `<b>${props.nam_en}</b><br />${props.KBAs2019} % of total land that is forest` : 'Hover over a state';
		this._div.innerHTML = `<h4>15.1.1 Forest area as a proportion of total land area</h4>${contents}`;
	};

	info.addTo(map);


	
	
	function getColor(d) {
		return d > 100 ? '#005a32' :
			d > 100  ? '#238b45' :
			d > 80  ? '#41ab5d' :
			d > 60  ? '#74c476' :
			d > 40   ? '#a1d99b' :
			d > 10   ? '#c7e9c0' :
			d > 0   ? '#e5f5e0' : '#f7fcf5';
	}

//update the fill color to your variable 
	function style(feature) {
		return {
			weight: 2,
			opacity: 1,
			color: 'white',
			dashArray: '3',
			fillOpacity: 0.7,
			fillColor: getColor(feature.properties.KBAs2019)
		};
	}

	function highlightFeature(e) {
		const layer = e.target;

		layer.setStyle({
			weight: 5,
			color: '#666',
			dashArray: '',
			fillOpacity: 0.7
		});

		layer.bringToFront();

		info.update(layer.feature.properties);
	}

	/* global statesData */ //UPDATE nationData
	const geojson = L.geoJson(nationData, {
		style,
		onEachFeature
	}).addTo(map);

	function resetHighlight(e) {
		geojson.resetStyle(e.target);
		info.update();
	}

	function zoomToFeature(e) {
		map.fitBounds(e.target.getBounds());
	}

	function onEachFeature(feature, layer) {
		layer.on({
			mouseover: highlightFeature,
			mouseout: resetHighlight,
			click: zoomToFeature
		});
	}

	map.attributionControl.addAttribution('SDG data &copy; <a href="https://unstats-undesa.opendata.arcgis.com/">UN SDG DataHub</a>');



	const legend = L.control({position: 'bottomright'});

	legend.onAdd = function (map) {

		const div = L.DomUtil.create('div', 'info legend');
		const grades = [0, 10, 20, 40, 60, 80, 100];
		const labels = [];
		let from, to;

		for (let i = 0; i < grades.length; i++) {
			from = grades[i];
			to = grades[i + 1];

			labels.push(`<i style="background:${getColor(from + 1)}"></i> ${from}${to ? `&ndash;${to}` : '+'}`);
		}

		labels.push(`<i style="background:#FFFFFF"></i>  No Data`);

		div.innerHTML = labels.join('<br>');
		return div;
	};

	legend.addTo(map);

</script>
